import numpy as np
import matplotlib.pyplot as plt

# Generate data
x = np.linspace(0, 2 * np.pi, 100)  # Create an array of 100 values from 0 to 2π
y = np.sin(x)  # Compute sine of each value

# Plot
plt.plot(x, y)
plt.title('Sine Wave from 0 to 2π')
plt.xlabel('x')
plt.ylabel('sin(x)')
plt.grid(True)
plt.show()
